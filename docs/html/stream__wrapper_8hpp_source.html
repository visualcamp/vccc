336 template<typename Rep, typename Period>
337 void BasicStreamWrapper<CharT, String, Stream>::write(const std::chrono::duration<Rep, Period>& duration_) {
338  IOSFlagsSaver<stream_type> saver(stream_);
339 
340  auto duration =
341  duration_ >= std::chrono::duration<Rep, Period>::zero() ?
342  duration_ : -duration_;
343 
344  if(duration < std::chrono::minutes(1)) {
345  auto ns = std::chrono::duration_cast<std::chrono::nanoseconds>(duration);
346  stream_ << std::setprecision(5);
347 
348  if(duration < std::chrono::milliseconds(1)) {
349  if(duration < std::chrono::nanoseconds(1'000))
350  stream_ << ns.count() << "ns";
351  else
352  stream_ << ns.count() / 1'000.L << "us";
353  }
354  else { // duration >= std::chrono::milliseconds(1)
355  if(duration < std::chrono::milliseconds(1'000))
356  stream_ << ns.count() / 1'000'000.L << "ms";
357  else
358  stream_ << ns.count() / 1'000'000'000.L << "s";
359  }
360  }
361  else {
362  using days = std::chrono::duration<long, std::ratio< 86400>>;
363  using months = std::chrono::duration<long, std::ratio< 2629746>>;
364  using years = std::chrono::duration<long, std::ratio<31556952>>;
365 
366  auto hh = std::chrono::duration_cast<std::chrono::hours >(duration % days (1)).count();
367  auto mm = std::chrono::duration_cast<std::chrono::minutes >(duration % std::chrono::hours (1)).count();
368  auto ss = std::chrono::duration_cast<std::chrono::seconds >(duration % std::chrono::minutes(1)).count();
369  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(duration % std::chrono::seconds(1)).count();
370 
371  if(duration >= days(1)) {
372  auto DD = std::chrono::duration_cast<days >(duration % months(1)).count();
373  auto MM = std::chrono::duration_cast<months>(duration % years(1)).count();
374  auto YY = std::chrono::duration_cast<years >(duration).count();
375 
376  if(YY) stream_ << YY << "Y ";
377  if(MM) stream_ << MM << "M ";
378  if(DD) stream_ << DD << "D ";
379  stream_ << std::setw(2) << std::setfill(' ');
380  }
381 
382  if(hh || mm || ss || ms) {
383  if (duration >= std::chrono::hours(1))
384  stream_ << hh << "h "
385  << std::setw(2) << std::setfill(' ') << mm << "m ";
386  else if (duration >= std::chrono::minutes(1))
387  stream_ << mm << "m ";
388 
389  stream_ << std::setw(2) << std::setfill(' ') << ss;
390  if (ms) stream_ << '.' << std::setw(3) << std::setfill('0') << ms;
391  stream_ << 's';
392  }
393  }
