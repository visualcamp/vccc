45convert_to_impl(std::true_type, const T& from, std::index_sequence<I...> index_seq)

47 ENSURES(check_bound<std::tuple_size<R>::value>(bool_constant<!is_tuple_like<T>::value && has_size<T>::value>{}, from),
48 "Following must be satisfied: from.size() >= tuple_size<R>::value");
49 using Zeros = make_zero_sequence<(

52 return fill_rest<R>(from, index_seq, Zeros{});
