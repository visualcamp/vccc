264 void swap(optional& other)
265#if __cplusplus >= 201703
266 noexcept(std::is_nothrow_move_constructible<value_type>::value && std::is_nothrow_swappable<value_type>::value)
267#else
268 noexcept(std::is_nothrow_move_constructible<value_type>::value)
269#endif
270 {
271 static_assert(std::is_move_constructible<value_type>::value,
272 "vccc::optional<T>::swap : T must be move constructible");
273
274 if (other.has_value()) {
275 if (this->has_value()) {
276 std::swap(**this, *other);
277 } else { // !this->has_value()
278 this->construct(std::move(*other));
279 other.reset();
280 }
281 } else {
282 if (this->has_value()) {
283 other.construct(**this);
